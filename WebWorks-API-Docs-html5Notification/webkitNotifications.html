function supportNotification() {

checkPermission

Number checkPermission()


Supported Platform(s)

 - BlackBerry PlayBook 1.0+
 - BlackBerry 10
 - Ripple Emulator

Description

This function will check Permission.



Returns

checkedPermission The permission to access the notification



Code Example(s)

 
function checkNotificationPermission() {
  if (window.webkitNotifications.checkPermission() != 0) { // 0 is PERMISSION_ALLOWED
    window.webkitNotifications.requestPermission();
  }
}

show

void show()


Supported Platform(s)

 - BlackBerry PlayBook 1.0+
 - BlackBerry 10
 - Ripple Emulator

Description

This function will show web notification.



Code Example(s)

function showWebNotification() {
  var url = 'http://chart.apis.google.com/chart?cht=bvg&chs=150x90&chd=s:Monkeys&chxt=x,y&chxs=0,ff0000,12,0,lt|1,0000ff,10,1,lt';
  if (window.webkitNotifications.checkPermission() == 0) {
    window.webkitNotifications.createHTMLNotification(url).show();
  } else {
    alert("this page doesn't have permission to show notification");
  }
}
